### Get your socks on!

`react-socks` is here to help

```jsx
// App.js
import { BreakpointProvider, setDefaultBreakpoints } from 'react-socks';

// Set these once
setDefaultBreakpoints([
    { mobile: 0 },
    { tablet: 401 },
    { desktop: 769}
]);

// Wrap the app
const App = () => (
    <BreakpointProvider>
        <MySuperDuperApp />
    </BreakpointProvider>
);
```
